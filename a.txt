    <script>
        let entries = [];
        let editIndex = null;

        const quotes = [
            "Believe you can and you're halfway there.",
            "You are enough just as you are.",
            "The best way to predict the future is to create it.",
            "Every day is a fresh start.",
            "Turn your wounds into wisdom."
        ];

        function nextQuote() {
            const el = document.getElementById("quote");
            el.style.opacity = 0;
            setTimeout(() => {
                el.innerText = `"${quotes[Math.floor(Math.random() * quotes.length)]}"`;
                el.style.opacity = 1;
            }, 300);
        }

        async function loadEntries() {
            try {
                const res = await fetch("../php/get_entries.php");
                if (res.status === 403) return window.location.href = "login.php";
                const data = await res.json();
                entries = data;
                document.getElementById("totalEntries").innerText = entries.length;
                renderEntries();
            } catch (e) {
                console.error(e);
            }
        }

        function renderEntries() {
            const list = document.getElementById("entries-list");
            list.innerHTML = "";

            if (entries.length === 0) {
                list.innerHTML = `
                <div class="empty-state">
                    <i class="fa-regular fa-paper-plane"></i>
                    <p>No entries yet. Start writing today!</p>
                </div>`;
                return;
            }

            entries.forEach((entry, i) => {
                const date = new Date(entry.entry_date).toLocaleDateString('en-US', { month: 'short', day: 'numeric', hour: 'numeric', minute: 'numeric' });
                const item = document.createElement("div");
                item.className = "entry-item";
                item.onclick = (e) => {
                    // Prevent edit/delete clicks from toggling
                    if (e.target.closest('button')) return;
                    // Optional: View details
                };
                item.innerHTML = `
                    <div class="entry-top">
                        <span class="entry-date">${date}</span>
                        <div class="entry-actions-mini" style="display:flex; gap:8px;">
                            <button onclick="openEdit(${i})" style="border:none; background:none; cursor:pointer; color:var(--text-muted);"><i class="fa-solid fa-pen"></i></button>
                            <button onclick="deleteEntry(${i})" style="border:none; background:none; cursor:pointer; color:#ef4444;"><i class="fa-solid fa-trash"></i></button>
                        </div>
                    </div>
                    <div style="display:flex; align-items:center; gap:8px; margin-bottom:8px;">
                        <span class="entry-mood">${getMoodIcon(entry.mood)}</span>
                        <strong style="font-size:0.9rem;">${entry.mood}</strong>
                    </div>
                    <p class="entry-preview">${entry.text}</p>
                `;
                list.appendChild(item);
            });
        }

        function getMoodIcon(mood) {

        async function saveEntry() {
            const text = document.getElementById("thoughts").value;
            const mood = document.getElementById("mood").value;

            const fd = new FormData();
            fd.append("text", text);
            fd.append("mood", mood);
            await fetch("../php/save_entry.php", { method: "POST", body: fd });
            document.getElementById("thoughts").value = "";
            loadEntries();

            fd.append("id", entries[i].id);
            await fetch("../php/delete_entry.php", { method: "POST", body: fd });
            loadEntries();
